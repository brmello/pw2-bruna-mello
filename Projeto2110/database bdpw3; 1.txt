create database bdpw3;

use bdpw3;


create table demo( 
    ID INT Auto_increment primary key,
    nome varchar (80),
    idade int (3),
    cpf varchar (11),
    endereco varchar (250),
    email varchar (100),
    celular int (15)
    );

create table usuarios (
    id int UNSIGNED ZEROFILL NOT NULL AUTO_INCREMENT,
    login varchar(30),
    senha varchar(40),
    primary key (id)) ENGINE = MyISAM;
